```
export const REG_EXP = {
  mobile: /^1[3-9]\d{9}$/,
  // 密码6-16位，至少包含数字、字母（区分大小写）、符号中的2种，且不能有空格
  password: /^(?![a-zA-Z]+$)(?![\d]+$)(?![\W_]+$)[^\s]{6,16}$/,
  email: /\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*/,
  idNumber: /^(^[1-9]\d{7}((0\d)|(1[0-2]))(([0|1|2]\d)|3[0-1])\d{3}$)|(^[1-9]\d{5}[1-9]\d{3}((0\d)|(1[0-2]))(([0|1|2]\d)|3[0-1])((\d{4})|\d{3}[Xx])$)$/,
  phone: /^[-\d]+$/,
  blank: /^[\s]+$/,
  nick: /^[A-Za-z0-9\u4e00-\u9fa5]{1,16}$/ // 用户名，数字、汉字和字母
}
```
